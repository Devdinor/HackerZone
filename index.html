server.py

import socket HOST = '127.0.0.1' # Alamat standar (localhost) PORT = 65431 # Port untuk didengarkan (port non-privilege adalah > 1023) with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s: s.bind((HOST, PORT)) s.listen() conn, addr = s.accept() with conn: print('Terhubung dengan', addr) while True: data = conn.recv(1024) if not data: break conn.sendall(data)

#!/usr/bin/env python3 import socket HOST = '127.0.0.1' # localhost nama server PORT = 65431 # port yang dipake di server with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s: s.connect((HOST, PORT)) s.sendall(b'Hello, world') data = s.recv(1024) print('Received', repr(data))
